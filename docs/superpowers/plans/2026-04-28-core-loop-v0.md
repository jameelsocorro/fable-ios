```swift
import Foundation

enum CalendarDateKey {
    static func key(for date: Date, calendar: Calendar = .autoupdatingCurrent) -> String {
        let components = calendar.dateComponents([.year, .month, .day], from: date)
        let year = components.year ?? 0
        let month = components.month ?? 0
        let day = components.day ?? 0
        return String(format: "%04d-%02d-%02d", year, month, day)
    }

    static func date(from key: String, calendar: Calendar = .autoupdatingCurrent) -> Date? {
        let parts = key.split(separator: "-").compactMap { Int($0) }
        guard parts.count == 3 else { return nil }
        return calendar.date(from: DateComponents(year: parts[0], month: parts[1], day: parts[2]))
    }

    static func dayOffset(for date: Date, calendar: Calendar = .autoupdatingCurrent) -> Int {
        let startOfDay = calendar.startOfDay(for: date)
        let reference = calendar.date(from: DateComponents(year: 2001, month: 1, day: 1))!
        return calendar.dateComponents([.day], from: reference, to: startOfDay).day ?? 0
    }

    static func daysBetween(_ startKey: String, _ endKey: String, calendar: Calendar = .autoupdatingCurrent) -> Int? {
        guard
            let startDate = date(from: startKey, calendar: calendar),
            let endDate = date(from: endKey, calendar: calendar)
        else {
            return nil
        }

        return calendar.dateComponents([.day], from: startDate, to: endDate).day
    }
}
```

Create `Fable/Fable/Quests/DailyQuestSelector.swift`:

```swift
import Foundation

struct DailyQuestSelector {
    var calendar: Calendar

    init(calendar: Calendar = .autoupdatingCurrent) {
        self.calendar = calendar
    }

    func selectDefinitions(
        for date: Date,
        catalog: [QuestDefinition] = QuestCatalog.definitions
    ) -> [QuestDefinition] {
        let dayOffset = CalendarDateKey.dayOffset(for: date, calendar: calendar)

        return QuestType.allCases.enumerated().compactMap { categoryIndex, questType in
            let definitions = catalog
                .filter { $0.type == questType }
                .sorted { $0.id < $1.id }

            guard !definitions.isEmpty else { return nil }

            let selectedIndex = abs(dayOffset + categoryIndex) % definitions.count
            return definitions[selectedIndex]
        }
    }
}
```

```swift
        self.localDateKey = localDateKey
        self.questDefinitionID = questDefinitionID
        self.questTypeRawValue = questType.rawValue
        self.renderedTitle = renderedTitle
        self.renderedDescription = renderedDescription
        self.renderedFallbackPrompt = renderedFallbackPrompt
        self.renderedDraftTemplate = renderedDraftTemplate
        self.suggestedPlatform = suggestedPlatform
        self.displayOrder = displayOrder
        self.statusRawValue = status.rawValue
        self.completedAt = completedAt
        self.skippedAt = skippedAt
        self.copiedDraftAt = copiedDraftAt
        self.createdAt = createdAt
    }

    var questType: QuestType {
        QuestType(rawValue: questTypeRawValue) ?? .shareProgress
    }

    var status: DailyQuestStatus {
        get { DailyQuestStatus(rawValue: statusRawValue) ?? .available }
        set { statusRawValue = newValue.rawValue }
    }

    var isCompleted: Bool {
        status == .completed
    }

    func markCompleted(at date: Date) {
        status = .completed
        completedAt = date
        skippedAt = nil
    }

    func markSkipped(at date: Date) {
        guard status != .completed else { return }
        status = .skipped
        skippedAt = date
    }

    func markDraftCopied(at date: Date) {
        copiedDraftAt = date
    }
}
```

```swift
import Foundation
import SwiftData

@MainActor
struct DailyQuestMaterializer {
    var calendar: Calendar
    var selector: DailyQuestSelector

    init(calendar: Calendar = .autoupdatingCurrent) {
        self.calendar = calendar
        self.selector = DailyQuestSelector(calendar: calendar)
    }

    func materializeQuests(
        for date: Date = .now,
        profile: FounderProfile,
        in context: ModelContext
    ) throws -> [DailyQuest] {
        let dateKey = CalendarDateKey.key(for: date, calendar: calendar)
        let existing = try existingQuests(for: dateKey, in: context)

        if !existing.isEmpty {
            return existing.sorted { $0.displayOrder < $1.displayOrder }
        }

        let selectedDefinitions = selector.selectDefinitions(for: date)
        let renderedQuests = selectedDefinitions.enumerated().map { index, definition in
            DailyQuest(
                localDateKey: dateKey,
                questDefinitionID: definition.id,
                questType: definition.type,
                renderedTitle: QuestTemplateRenderer.render(definition.titleTemplate, profile: profile.templateDefaults),
                renderedDescription: QuestTemplateRenderer.render(definition.descriptionTemplate, profile: profile.templateDefaults),
                renderedFallbackPrompt: QuestTemplateRenderer.render(definition.fallbackPrompt, profile: profile.templateDefaults),
                renderedDraftTemplate: QuestTemplateRenderer.render(definition.draftTemplate, profile: profile.templateDefaults),
                suggestedPlatform: definition.suggestedPlatform,
                displayOrder: index
            )
        }

        for quest in renderedQuests {
            context.insert(quest)
        }

        try context.save()
        return renderedQuests
    }

    private func existingQuests(for dateKey: String, in context: ModelContext) throws -> [DailyQuest] {
        let descriptor = FetchDescriptor<DailyQuest>(
            predicate: #Predicate<DailyQuest> { quest in
                quest.localDateKey == dateKey
            },
            sortBy: [SortDescriptor(\DailyQuest.displayOrder)]
        )
        return try context.fetch(descriptor)
    }
}
```

```swift
import Foundation

struct StreakEvaluation: Equatable {
    var currentStreak: Int
    var longestStreak: Int
    var lastCompletedDateKey: String?
    var phase: StreakPhase
}

struct StreakCalculator {
    var calendar: Calendar

    init(calendar: Calendar = .autoupdatingCurrent) {
        self.calendar = calendar
    }

    func evaluate(
        completedDateKeys: [String],
        today: Date,
        previousLongestStreak: Int
    ) -> StreakEvaluation {
        let uniqueKeys = Array(Set(completedDateKeys))
        let sortedKeys = uniqueKeys
            .compactMap { key -> (key: String, date: Date)? in
                guard let date = CalendarDateKey.date(from: key, calendar: calendar) else { return nil }
                return (key, date)
            }
            .filter { $0.date <= calendar.startOfDay(for: today) }
            .sorted { $0.date < $1.date }

        guard let latest = sortedKeys.last else {
            return StreakEvaluation(
                currentStreak: 0,
                longestStreak: previousLongestStreak,
                lastCompletedDateKey: nil,
                phase: .reset
            )
        }

        let daysSinceLatest = calendar.dateComponents(
            [.day],
            from: latest.date,
            to: calendar.startOfDay(for: today)
        ).day ?? 0

        if daysSinceLatest >= 3 {
            return StreakEvaluation(
                currentStreak: 0,
                longestStreak: previousLongestStreak,
                lastCompletedDateKey: latest.key,
                phase: .reset
            )
        }

        let currentStreak = completedDayStreak(endingAt: latest, in: sortedKeys)
        let phase: StreakPhase = daysSinceLatest == 2 ? .grace : .active

        return StreakEvaluation(
            currentStreak: currentStreak,
            longestStreak: max(previousLongestStreak, currentStreak),
            lastCompletedDateKey: latest.key,
            phase: phase
        )
    }

    private func completedDayStreak(
        endingAt latest: (key: String, date: Date),
        in sortedKeys: [(key: String, date: Date)]
    ) -> Int {
        var streak = 1
        var cursor = latest.date

        for entry in sortedKeys.dropLast().reversed() {
            let gap = calendar.dateComponents([.day], from: entry.date, to: cursor).day ?? Int.max
            if gap == 1 || gap == 2 {
                streak += 1
                cursor = entry.date
            } else if gap >= 3 {
                break
            }
        }

        return streak
    }
}
```

```swift
struct QuestProgressService {
    var calendar: Calendar

    init(calendar: Calendar = .autoupdatingCurrent) {
        self.calendar = calendar
    }

    func complete(_ quest: DailyQuest, at date: Date = .now, in context: ModelContext) throws {
        quest.markCompleted(at: date)
        try refreshStreakSnapshot(today: date, in: context)
        try context.save()
    }

    func skip(_ quest: DailyQuest, at date: Date = .now, in context: ModelContext) throws {
        quest.markSkipped(at: date)
        try context.save()
    }

    func markDraftCopied(_ quest: DailyQuest, at date: Date = .now, in context: ModelContext) throws {
        quest.markDraftCopied(at: date)
        try context.save()
    }

    func refreshStreakSnapshot(today: Date = .now, in context: ModelContext) throws {
        let completedRawValue = DailyQuestStatus.completed.rawValue
        let completedDescriptor = FetchDescriptor<DailyQuest>(
            predicate: #Predicate<DailyQuest> { quest in
                quest.statusRawValue == completedRawValue
            }
        )
        let completedDateKeys = try context.fetch(completedDescriptor).map(\.localDateKey)
        let snapshot = try existingSnapshot(in: context)

        let evaluation = StreakCalculator(calendar: calendar).evaluate(
            completedDateKeys: completedDateKeys,
            today: today,
            previousLongestStreak: snapshot.longestStreak
        )
        snapshot.apply(evaluation, updatedAt: today)
    }

    private func existingSnapshot(in context: ModelContext) throws -> StreakSnapshot {
        if let snapshot = try context.fetch(FetchDescriptor<StreakSnapshot>()).first {
            return snapshot
        }

        let snapshot = StreakSnapshot()
        context.insert(snapshot)
        return snapshot
    }
```

```swift
import SwiftData
import SwiftUI
import UIKit

struct TodayView: View {
    @Environment(\.modelContext) private var modelContext
    @Query(sort: \DailyQuest.displayOrder) private var allDailyQuests: [DailyQuest]
    @Query private var snapshots: [StreakSnapshot]

    @State private var today = Date()
    @State private var errorMessage: String?
    @State private var copiedQuestID: UUID?
    @State private var completionFeedbackTrigger = 0

    private var calendar: Calendar {
        .autoupdatingCurrent
    }

    private var todayKey: String {
        CalendarDateKey.key(for: today, calendar: calendar)
    }

    private var todaysQuests: [DailyQuest] {
        allDailyQuests
            .filter { $0.localDateKey == todayKey }
            .sorted { $0.displayOrder < $1.displayOrder }
    }

    private var snapshot: StreakSnapshot? {
        snapshots.first
    }

    var body: some View {
        ScrollView {
            VStack(alignment: .leading, spacing: 20) {
                header

                if let errorMessage {
                    errorState(errorMessage)
                } else if todaysQuests.isEmpty {
                    loadingState
                } else {
                    VStack(spacing: 16) {
                        ForEach(todaysQuests) { quest in
                            QuestCardView(
                                quest: quest,
                                didCopyDraft: copiedQuestID == quest.id,
                                complete: { complete(quest) },
                                skip: { skip(quest) },
                                copyDraft: { copyDraft(for: quest) }
                            )
                        }
                    }
                }
            }
            .padding(.horizontal, 20)
            .padding(.vertical, 24)
        }
        .accessibilityIdentifier("today.scroll")
        .background(background)
        .task {
            bootstrapToday()
        }
        .sensoryFeedback(.success, trigger: completionFeedbackTrigger)
    }

    private var header: some View {
        VStack(alignment: .leading, spacing: 16) {
            VStack(alignment: .leading, spacing: 6) {
                Text("Today")
                    .font(.largeTitle.weight(.semibold))
                    .foregroundStyle(FableColors.lightTextPrimary)

                Text("Complete one quest to keep your story visible.")
                    .font(.subheadline)
                    .foregroundStyle(FableColors.lightTextMuted)
            }

            StreakSummaryView(snapshot: snapshot)
        }
    }

    private var loadingState: some View {
        VStack(alignment: .leading, spacing: 8) {
            ProgressView()
            Text("Preparing today's quests.")
                .font(.subheadline)
                .foregroundStyle(FableColors.lightTextMuted)
        }
        .fableCardSurface()
    }

    private func errorState(_ message: String) -> some View {
        VStack(alignment: .leading, spacing: 10) {
            Text("Today is unavailable")
                .font(.headline)
                .foregroundStyle(FableColors.lightTextPrimary)

            Text(message)
                .font(.subheadline)
                .foregroundStyle(FableColors.lightTextMuted)

            Button("Try again", systemImage: "arrow.clockwise", action: bootstrapToday)
                .buttonStyle(.borderedProminent)
                .tint(FableColors.brandPrimary)
        }
        .fableCardSurface()
    }

    private var background: some View {
        LinearGradient(
            colors: [
                FableColors.lightBackground,
                FableColors.lightSurface
            ],
            startPoint: .top,
            endPoint: .bottom
        )
        .ignoresSafeArea()
    }

    private func bootstrapToday() {
        do {
            today = Date()
            try AppBootstrapper.seedProfileAndSnapshot(in: modelContext)
            let profile = try currentProfile()
            try DailyQuestMaterializer(calendar: calendar)
                .materializeQuests(for: today, profile: profile, in: modelContext)
            try QuestProgressService(calendar: calendar)
                .refreshStreakSnapshot(today: today, in: modelContext)
            try modelContext.save()
            errorMessage = nil
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    private func currentProfile() throws -> FounderProfile {
        if let profile = try modelContext.fetch(FetchDescriptor<FounderProfile>()).first {
            return profile
        }

        let profile = FounderProfile()
        modelContext.insert(profile)
        try modelContext.save()
        return profile
    }

    private func complete(_ quest: DailyQuest) {
        do {
            try QuestProgressService(calendar: calendar).complete(quest, at: Date(), in: modelContext)
            completionFeedbackTrigger += 1
            errorMessage = nil
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    private func skip(_ quest: DailyQuest) {
        do {
            try QuestProgressService(calendar: calendar).skip(quest, at: Date(), in: modelContext)
            errorMessage = nil
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    private func copyDraft(for quest: DailyQuest) {
        UIPasteboard.general.string = quest.renderedDraftTemplate
        copiedQuestID = quest.id

        do {
            try QuestProgressService(calendar: calendar).markDraftCopied(quest, at: Date(), in: modelContext)
            errorMessage = nil
        } catch {
            errorMessage = error.localizedDescription
        }
    }
}
```
